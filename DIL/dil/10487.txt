						Cookies.set('confirmed', 'true', { expires: 7000 });
						$('.cookiefooter').slideToggle();
					});
				});
						
			$(document).ready( function() {
				$('.navswitcher').on('click', function() {
					if ($(this).hasClass('full')) {
						Cookies.set('fullmode', 'true',{ expires: 7000 });
					}
					else {
						Cookies.remove('fullmode');
					}
					location.reload();
				});
			});
