        <span class="def">great</span> <span class="def">knowledge</span> ; <span class="def">poetic</span> <span class="def">talent</span>, <span class="def">inspiration</span> ; <span class="def">fore-knowledge</span> ; <span class="def">magic</span> <span class="def">lore</span>:</span>

          <span><strong class="pop" style="color:blue" href="bibliography/LL" title="eDIL Bibliography">LL</strong></span> 30d34
