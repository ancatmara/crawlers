    <p class="result">
      <span class="pos n" style="color: #ff8000">n</span>  perhaps originally <span class="trans">
        <span class="def">preparation</span>, <span class="def">time</span> of preparation</span> 
(<span class="pos vn." style="color: #ff8000">vn.</span> of <span class="ref"><a class="normal" href="search?search_in=headword&amp;q=fo-fera">fo-fera</a></span>, <span class="lbl">cf.</span>, <span class="ref"><a class="normal" href="search?search_in=headword&amp;q=fuar">fuar</a></span>, <span class="ref"><a class="normal" href="search?search_in=headword&amp;q=1 f&#xFA;r">1 fúr</a></span>), hence transf.</p>
